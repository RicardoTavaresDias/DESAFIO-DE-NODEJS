## Instalações ##

npm i drizzle-orm pg
npm i drizzle-kit -D
docker compose up -d

# Tipagem do zod no fastify
  npm i zod fastify-type-provider-zod

# swagger no fastify
  npm i @fastify/swagger

# layaut padão do swagger
  npm i @fastify/swagger-ui 

# layaut do swagger
  npm install @scalar/fastify-api-reference

# Para Seed banco de dados
  npm i @faker-js/faker -D

# Testes unitario
  npm i vitest -D
  npm i supertest -D
  npm i @types/supertest -D

# Para testes conseguir ver as variaveis de ambiente
  npm i dotenv-cli -D

# Matar o docker
  docker compose down

# Gera o grafico no teste no terminal, fica mais facil leitura.
  npm i -D @vitest/coverage-v8
